*** QUANTOS NOMES e SOBRENOMES UNICOS TEMOS NA TABELA PERSON?

SELECT DISTINCT fistname, lastname
FROM person.person

****A EQUIPE DE PRODUÇAO DE PRODUTOS PRECISA DO NOME DE TODAS AS PEÇAS QUE PESAM
MAIS QUE 500 KG MAS NAO MAIS QUE 700 KG****

SELECT Name
FROM Production.Product
WHERE Weight > 500 and Weight < 700


**** FOI PEDIDO PELO MARKETING UMA RELAÇAO DE TODOS OS EMPREGADOS(EMPLOYEES)
QUE SAO CASADOS (SINGLE = SOLTEIRO , MARRIED = CASADO) E SAO ASALARIADOS(SALARIED)


SELECT *
FROM HumanResources.Employee
WHERE MaritalStatus ='m' and SalariedFlag = 'true'
